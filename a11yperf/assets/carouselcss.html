.merry-go-round {
  display: flex;
  align-items: center;
  justify-content: center;
  width: 280px;
  margin: auto;
  ul {
    margin: 0 20px;
    display: flex;
  }
  li {
    display: inline-block;
    flex: 0 0 164px;
    transition: left 200ms;
    position: relative;
    left: 0;
  }
$count: 0;
$itemwidth: 164px;
@while $count < 20 {
  .left#{$count} li {
    left: $count * $itemwidth * -1
  }
  $count: $count + 1
}

function mgrChange (name, value) {
  var ul = document.querySelector('.' + name);
  ul.setAttribute('class', 
                      name + ' left' + value);
}
